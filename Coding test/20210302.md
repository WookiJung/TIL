- 반복문지우기 by 이현경
```python
# 반복문 지우기


for tc in range(int(input())):
    string = input()

    # stack 초기화 문자수는 1000이내
    stack = [0] * 1000

    # top 초기화
    top = -1

    # 문자열을 순회하면서 각 글자를 stack에 쌓기
    for s in string:
        top += 1
        stack[top] = s

        # 중복 검사(두개 이상 쌓였을 때부터)
        if top > 0:
            if stack[top] == stack[top-1]:
                # 중복된 두개 빼기
                stack[top] = 0
                top -= 1
                stack[top] = 0
                top -= 1

    cnt = 0
    for i in range(len(stack)):
        if stack[i] != 0:
            cnt += 1

    print('#{} {}'.format(tc+1, cnt))

```
